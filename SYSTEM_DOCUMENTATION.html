System Purpose

Mtokoma Motorcycle Parts is a full-featured inventory management system designed for motorcycle parts dealers. It handles the complete lifecycle from purchasing stock, managing inventory, processing sales, tracking expenses, and generating comprehensive financial reports.

Core Capabilities

  • Purchase management with automatic stock intake
  • Point-of-sale with EFD-format thermal receipt printing
  • Real-time stock tracking with movement audit trail
  • Stock reconciliation (system vs physical count)
  • Expense tracking with receipt attachment
  • Role-based permissions with dynamic UI hiding
  • CSV/Excel import and export across all modules
  • Email notifications on every transaction
  • Comprehensive reporting with PDF/CSV export
  • Autofills and automations to minimize clicks

Database Schema

Database name: mtokoma_db - Engine: InnoDB - Charset: utf8mb4

Key Principle: Stock balance is never stored on the items table. It is always calculated from stock_movements to maintain a complete audit trail.

Stock Auto-Update Logic

When Purchase is Saved as Received

DB::transaction(function () use ($data) {
    $purchase = Purchase::create($data);

    foreach ($data['items'] as $item) {
        PurchaseItem::create([
            'purchase_id' => $purchase->id,
            'item_id'     => $item['item_id'],
            'qty'         => $item['qty'],
            'unit_cost'   => $item['unit_cost'],
            'subtotal'    => $item['qty'] * $item['unit_cost'],
        ]);

        $lastBalance = StockMovement::where('item_id', $item['item_id'])
                         ->latest('id')->value('balance_after') ?? 0;

        $newBalance = $lastBalance + $item['qty'];

        StockMovement::create([
            'item_id'       => $item['item_id'],
            'type'          => 'purchase',
            'reference_id'  => $purchase->id,
            'reference_type'=> 'App\Models\Purchase',
            'qty_in'        => $item['qty'],
            'qty_out'       => 0,
            'balance_after' => $newBalance,
            'unit_cost'     => $item['unit_cost'],
            'user_id'       => auth()->id(),
        ]);
    }

    // Send email notification
    Mail::to(getManagers())->send(new PurchaseCreatedMail($purchase));
});

When Sale is Completed

DB::transaction(function () use ($data) {
    $sale = Sale::create($data);

    foreach ($data['items'] as $item) {
        SaleItem::create([
            'sale_id'    => $sale->id,
            'item_id'    => $item['item_id'],
            'qty'        => $item['qty'],
            'unit_price' => $item['unit_price'],
            'unit_cost'  => $item['unit_cost'],
            'subtotal'   => $item['qty'] * $item['unit_price'],
            'profit'     => ($item['qty'] * $item['unit_price']) - ($item['qty'] * $item['unit_cost']),
        ]);

        $lastBalance = StockMovement::where('item_id', $item['item_id'])
                         ->latest('id')->value('balance_after') ?? 0;

        $newBalance = $lastBalance - $item['qty'];

        StockMovement::create([
            'item_id'       => $item['item_id'],
            'type'          => 'sale',
            'reference_id'  => $sale->id,
            'reference_type'=> 'App\Models\Sale',
            'qty_in'        => 0,
            'qty_out'       => $item['qty'],
            'balance_after' => $newBalance,
            'unit_cost'     => $item['unit_cost'],
            'user_id'       => auth()->id(),
        ]);
    }

    // Handle credit sales
    if ($sale->payment_method === 'credit') {
        $customer = Customer::find($sale->customer_id);
        $customer->current_balance += $sale->total;
        $customer->save();
    }

    // Send email notification
    Mail::to(getManagers())->send(new SaleCompletedMail($sale));
});

When Purchase is Voided

DB::transaction(function () use ($purchase) {
    // Reverse stock for each item
    foreach ($purchase->purchaseItems as $item) {
        $lastBalance = StockMovement::where('item_id', $item->item_id)
                         ->latest('id')->value('balance_after') ?? 0;

        $newBalance = $lastBalance - $item->qty;

        StockMovement::create([
            'item_id'       => $item->item_id,
            'type'          => 'purchase',
            'reference_id'  => $purchase->id,
            'reference_type'=> 'App\Models\Purchase',
            'qty_in'        => 0,
            'qty_out'       => $item->qty,
            'balance_after' => $newBalance,
            'unit_cost'     => $item->unit_cost,
            'notes'         => 'Purchase voided - stock reversed',
            'user_id'       => auth()->id(),
        ]);
    }

    $purchase->update(['status' => 'cancelled']);
});

When Sale is Voided

DB::transaction(function () use ($sale) {
    // Reverse stock for each item
    foreach ($sale->saleItems as $item) {
        $lastBalance = StockMovement::where('item_id', $item->item_id)
                         ->latest('id')->value('balance_after') ?? 0;

        $newBalance = $lastBalance + $item->qty;

        StockMovement::create([
            'item_id'       => $item->item_id,
            'type'          => 'return',
            'reference_id'  => $sale->id,
            'reference_type'=> 'App\Models\Sale',
            'qty_in'        => $item->qty,
            'qty_out'       => 0,
            'balance_after' => $newBalance,
            'unit_cost'     => $item->unit_cost,
            'notes'         => 'Sale voided - stock restored',
            'user_id'       => auth()->id(),
        ]);
    }

    // Reverse credit if applicable
    if ($sale->payment_method === 'credit') {
        $customer = Customer::find($sale->customer_id);
        $customer->current_balance -= $sale->total;
        $customer->save();
    }

    $sale->update(['status' => 'voided']);
});
